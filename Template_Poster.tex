\begin{parblock}{Introduction}
	\alert{Align elements both horizontally and vertically!}

	\vspace{0.333333\baselineskip}
	\noindent We define various lengths and commands to facilitate both horizontal and vertical alignment of all elements.

	\vspace{0.666667\baselineskip}
	\noindent\alert{Horizontal alignment}

	\begin{description}
		\item[\code{\textbackslash leftmargin}] Horizontal position of the left border of the left-most text column: \code{40.5mm} by default.
		\item[\code{\textbackslash rightmargin}] Horizontal position of the right border of the right-most text column: \code{40.5mm} by default.
		\item[\code{\textbackslash numcols}] Number of columns on poster: \code{3}~by default (increase to \code{4} or \code{5} for poster in landscape orientation).
		\item[\code{\textbackslash colsep}] Space between columns (so-called ``gutter''): \code{15.5mm} by default.
		\item[\code{\textbackslash colwidth}] Column width: paper width (841~mm for A0 portrait) minus margins (${2 \times 40.5}$~mm) minus gutters (${2 \times 15.5}$~mm), by default 243~mm.
	\end{description}
	\sidenote{Test citations: \cite{Lisi1995}, \cite{Koszegi2013}.}
\end{parblock}
