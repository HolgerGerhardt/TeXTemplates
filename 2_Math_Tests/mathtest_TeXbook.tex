\begin{align*}
\begin{aligned}
\biggl(\int_{-\infty}^\infty e^{-x^2}\,\mathup{d}x\biggr)^2
&=\int_{-\infty}^\infty\int_{-\infty}^\infty e^{-(x^2+y^2)}\,\mathup{d}x\,\mathup{d}y\\
&=\int_0^{2\piup}\int_0^\infty e^{-r^2}\,\mathup{d}r\,\mathup{d}\theta\\
&=\int_0^{2\piup}\biggl(e^{-\frac{r^2}{2}}
\biggl|_{r=0}^{r=\infty}\,\biggr)\,\mathup{d}\theta\\
&=\piup.
\end{aligned}
\end{align*}
