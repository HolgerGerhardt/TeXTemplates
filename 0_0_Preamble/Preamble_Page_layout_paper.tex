\newcommand{\appendixformat}[1][\currentname]{%
	\titleformat{\section}[hang]%
		{}{\textscale{1.3}{\textsf{\textbf{\appendixname~\thesection\quad}}}}{0pt}{\textscale{1.3}{\textsf{\textbf{#1}}}}[]%
}
\AtBeginEnvironment{appendices}{\appendixformat}
